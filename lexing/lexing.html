static float START_ANGLES[4] = {     PI, 3*PI/2,    0, PI/2 };

int getQuadrantXY(int dx, int dy) {
    if (dx < 0) {
        if (dy < 0) {
            return 0;
        } else {
            return 3;
        }
    } else {
        if (dy < 0) {
            return 1;
        } else {
            return 2;
        }
    }
}
